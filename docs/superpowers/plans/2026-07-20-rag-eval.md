> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add built-in RAG retrieval evals for recall@k, precision@k, hit_rate@k, and MRR, and expose them through the existing eval runner, API, and backstage Eval panel.

**Architecture:** Keep RAG Eval deterministic and independent of live project data. Gold cases provide fixed retrieval reports first, while the metric function accepts the same `retrieve_vector_memory()` report shape so future query/chunk/retriever experiments can reuse the same evaluator.

**Tech Stack:** Python dataclasses and pytest for backend evals, FastAPI existing `/api/evals/builtin`, React/Vitest for frontend display.

Add to `backend/app/services/evaluation.py`:

```python
@dataclass(frozen=True)
class ExpectedRetrievedDocument:
    source: str
    source_id: str
    label: str


def evaluate_rag_retrieval(
    retrieval_report: dict[str, Any],
    expected_documents: list[ExpectedRetrievedDocument],
    top_k: int = 5,
    threshold: float = 0.8,
) -> dict[str, Any]:
    hits = list(retrieval_report.get("hits") or [])[:top_k]
    expected_by_id = {
        _retrieval_identity({"source": item.source, "source_id": item.source_id}): item
        for item in expected_documents
    }
    detected_ids: set[str] = set()
    first_rank: int | None = None

    for index, hit in enumerate(hits, start=1):
        identity = _retrieval_identity(hit)
        if identity not in expected_by_id:
            continue
        detected_ids.add(identity)
        if first_rank is None:
            first_rank = index

    detected = [item for identity, item in expected_by_id.items() if identity in detected_ids]
    missed = [item for identity, item in expected_by_id.items() if identity not in detected_ids]
    recall = _round_rate(len(detected), len(expected_documents))
    precision = _round_rate(len(detected), top_k)
    return {
        "metric": "rag_retrieval",
        "top_k": top_k,
        "detected": [item.label for item in detected],
        "missed": [item.label for item in missed],
        "detected_count": len(detected),
        "total_count": len(expected_documents),
        "recall_at_k": recall,
        "precision_at_k": precision,
        "hit_rate_at_k": 1.0 if detected else 0.0,
        "mrr": round(1 / first_rank, 6) if first_rank else 0.0,
        "threshold": threshold,
        "passed": recall >= threshold,
    }


def _retrieval_identity(hit: dict[str, Any]) -> str:
    return f"{hit.get('source', '')}:{hit.get('source_id', '')}"
```
